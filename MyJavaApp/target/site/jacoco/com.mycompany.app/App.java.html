package com.mycompany.app;

import java.util.ArrayList;
import java.util.Scanner;
import java.util.logging.Logger;

// A simple Todo class to represent a task
class Todo {
    private String task;
    private boolean isCompleted;

    public Todo(String task) {
        this.task = task;
        this.isCompleted = false;
    }

    public void markAsCompleted() {
        this.isCompleted = true;
    }

    public boolean isCompleted() {
        return isCompleted;
    }

    public String getTask() {
        return task;
    }

    @Override
    public String toString() {
        return (isCompleted ? "[âœ”] " : "[ ] ") + task;
    }
}

public class App {
    private static final Logger logger = Logger.getLogger(App.class.getName());
    private static final ArrayList<Todo> todoList = new ArrayList<>();
    private static final Scanner scanner = new Scanner(System.in);

    public static void main(String[] args) {
        logger.info("Todo App Started");
        boolean running = true;

        while (running) {
            displayMenu();
            int choice = getUserChoice();
            switch (choice) {
                case 1 -> addTodo();
                case 2 -> markTodoAsCompleted();
                case 3 -> displayTodos();
                case 4 -> {
                    logger.info("Exiting Todo App...");
                    System.out.println("Goodbye!");
                    running = false;
                }
                default -> System.out.println("Invalid choice. Please choose a valid option.");
            }
        }
    }

    // Displays the main menu
    private static void displayMenu() {
        System.out.println("\nTodo App Menu:");
        System.out.println("1. Add a new Todo");
        System.out.println("2. Mark a Todo as completed");
        System.out.println("3. View all Todos");
        System.out.println("4. Exit");
        System.out.print("Enter your choice: ");
    }

    // Gets user choice and handles input
    private static int getUserChoice() {
        int choice = -1;
        try {
            choice = Integer.parseInt(scanner.nextLine());
        } catch (NumberFormatException e) {
            logger.warning("Invalid input. Expected an integer.");
        }
        return choice;
    }

    // Adds a new todo item to the list
    private static void addTodo() {
        System.out.print("Enter a new task: ");
        String task = scanner.nextLine();
        todoList.add(new Todo(task));
        logger.info("Added new task: " + task);
        System.out.println("Task added successfully!");
    }

    // Marks a todo item as completed
    private static void markTodoAsCompleted() {
        if (todoList.isEmpty()) {
            System.out.println("No tasks available.");
            return;
        }
        displayTodos();
        System.out.print("Enter the number of the task to mark as completed: ");
        int taskIndex = getUserChoice();
        if (taskIndex > 0 && taskIndex <= todoList.size()) {
            todoList.get(taskIndex - 1).markAsCompleted();
            logger.info("Task marked as completed: " + todoList.get(taskIndex - 1).getTask());
            System.out.println("Task marked as completed.");
        } else {
            logger.warning("Invalid task number.");
            System.out.println("Invalid task number.");
        }
    }

    // Displays all tasks in the todo list
    private static void displayTodos() {
        if (todoList.isEmpty()) {
            System.out.println("No tasks to display.");
        } else {
            System.out.println("\nYour Todo List:");
            for (int i = 0; i < todoList.size(); i++) {
                System.out.println((i + 1) + ". " + todoList.get(i));
            }
        }
    }
}
